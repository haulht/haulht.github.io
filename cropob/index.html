rotate & crop feature

HTML

<!-- This wraps the whole cropob -->
<div id="cropob_crop"></div>
<!-- Button is used to download image from CropOB -->
<button type="button" onclick="downloadImgCropOBType2();">Download Image</button>

Javascript

// Init cropob crop object
var cropobCrop = $("#cropob_crop").croima({
    'image_url' :'1.JPG',
    'type_feature': 2
});
// Download image
function downloadImgCropOBType2(){
  if(!cropobCrop)
    return null;
  var data = cropobCrop.croima('getImageData');
  window.open(data);
}
